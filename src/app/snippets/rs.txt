use std::collections::HashMap;

#[derive(Debug, Clone)]
pub struct Config {
    pub version: u32,
    pub active: bool,
    pub metadata: HashMap<String, String>,
}

impl Config {
    pub fn new(version: u32) -> Self {
        Self {
            version,
            active: true,
            metadata: HashMap::new(),
        }
    }

    pub fn validate(&self) -> Result<(), String> {
        if self.version == 0 {
            return Err("Invalid version".to_string());
        }
        Ok(())
    }
}
