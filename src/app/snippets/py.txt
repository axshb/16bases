import json
from dataclasses import dataclass, field
from typing import Dict

@dataclass
class Config:
    version: int
    active: bool = True
    metadata: Dict[str, str] = field(default_factory=dict)

class ConfigLoader:
    def __init__(self, source: str):
        self.source = source

    def parse(self) -> Config:
        with open(self.source, 'r') as f:
            data = json.load(f)
            return Config(**data)
